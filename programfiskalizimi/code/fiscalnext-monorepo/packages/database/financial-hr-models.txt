// Payroll Models
model Employee {
  id       String  @id @default(uuid())
  tenantId String  @map("tenant_id")
  userId   String? @map("user_id") // Link to User if they have system access

  // Personal Info
  firstName   String    @map("first_name")
  lastName    String    @map("last_name")
  email       String?
  phone       String?
  dateOfBirth DateTime? @map("date_of_birth")

  // Employment
  employeeNumber  String?   @unique @map("employee_number")
  department      String?
  position        String?
  hireDate        DateTime  @map("hire_date")
  terminationDate DateTime? @map("termination_date")
  employmentType  String    @default("full_time") @map("employment_type") // full_time, part_time, contract

  // Salary
  baseSalary       Decimal @map("base_salary") @db.Decimal(10, 2)
  currency         String  @default("ALL")
  paymentFrequency String  @default("monthly") @map("payment_frequency") // monthly, biweekly, weekly

  // Bank Details
  bankName    String? @map("bank_name")
  bankAccount String? @map("bank_account")

  // Tax Info
  taxId          String? @map("tax_id")
  socialSecurity String? @map("social_security")

  // Status
  isActive Boolean @default(true) @map("is_active")

  // Timestamps
  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  // Relations
  tenant              Tenant                @relation(fields: [tenantId], references: [id])
  payrollRuns         PayrollRunItem[]
  timeOffRequests     TimeOffRequest[]
  performanceReviews  PerformanceReview[]
  trainingEnrollments TrainingEnrollment[]
  documents           EmployeeDocument[]

  @@index([tenantId])
  @@index([employeeNumber])
  @@map("employees")
}

model PayrollRun {
  id       String @id @default(uuid())
  tenantId String @map("tenant_id")

  // Period
  periodStart DateTime @map("period_start")
  periodEnd   DateTime @map("period_end")
  payDate     DateTime @map("pay_date")

  // Status
  status String @default("draft") // draft, approved, paid, cancelled

  // Totals
  totalGross      Decimal @map("total_gross") @db.Decimal(10, 2)
  totalDeductions Decimal @map("total_deductions") @db.Decimal(10, 2)
  totalNet        Decimal @map("total_net") @db.Decimal(10, 2)

  // Approval
  approvedBy String?   @map("approved_by")
  approvedAt DateTime? @map("approved_at")

  // Payment
  paidBy String?   @map("paid_by")
  paidAt DateTime? @map("paid_at")

  notes String?

  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  // Relations
  tenant Tenant           @relation(fields: [tenantId], references: [id])
  items  PayrollRunItem[]

  @@index([tenantId])
  @@index([status])
  @@index([periodStart])
  @@map("payroll_runs")
}

model PayrollRunItem {
  id           String @id @default(uuid())
  payrollRunId String @map("payroll_run_id")
  employeeId   String @map("employee_id")

  // Gross Pay
  basePay  Decimal @map("base_pay") @db.Decimal(10, 2)
  overtime Decimal @default(0) @db.Decimal(10, 2)
  bonuses  Decimal @default(0) @db.Decimal(10, 2)
  grossPay Decimal @map("gross_pay") @db.Decimal(10, 2)

  // Deductions
  incomeTax       Decimal @default(0) @map("income_tax") @db.Decimal(10, 2)
  socialSecurity  Decimal @default(0) @map("social_security") @db.Decimal(10, 2)
  healthInsurance Decimal @default(0) @map("health_insurance") @db.Decimal(10, 2)
  otherDeductions Decimal @default(0) @map("other_deductions") @db.Decimal(10, 2)
  totalDeductions Decimal @map("total_deductions") @db.Decimal(10, 2)

  // Net Pay
  netPay Decimal @map("net_pay") @db.Decimal(10, 2)

  // Payslip
  payslipUrl String? @map("payslip_url")

  createdAt DateTime @default(now()) @map("created_at")

  // Relations
  payrollRun PayrollRun @relation(fields: [payrollRunId], references: [id], onDelete: Cascade)
  employee   Employee   @relation(fields: [employeeId], references: [id])

  @@index([payrollRunId])
  @@index([employeeId])
  @@map("payroll_run_items")
}

model BankTransaction {
  id            String @id @default(uuid())
  bankAccountId String @map("bank_account_id")

  // Transaction Details
  transactionDate DateTime @map("transaction_date")
  description     String
  reference       String?

  // Amount
  debit   Decimal? @db.Decimal(10, 2)
  credit  Decimal? @db.Decimal(10, 2)
  balance Decimal  @db.Decimal(10, 2)

  // Matching
  isMatched   Boolean @default(false) @map("is_matched")
  matchedWith String? @map("matched_with") // transaction_id or expense_id
  matchedType String? @map("matched_type") // transaction, expense, bill_payment

  // Import
  importedFrom String?   @map("imported_from") // csv, api
  importedAt   DateTime? @map("imported_at")

  createdAt DateTime @default(now()) @map("created_at")

  // Relations
  bankAccount BankAccount @relation(fields: [bankAccountId], references: [id], onDelete: Cascade)

  @@index([bankAccountId])
  @@index([transactionDate])
  @@index([isMatched])
  @@map("bank_transactions")
}
